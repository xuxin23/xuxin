对象私有方法
var test = function() {
alert(1);
var clas = new Classroom(["kobe","wade"],"jackson");
alert(2);
/*不能调用对象的私有方法*/
// clas.disp();
}

//表示教室的一个对象构造函数
function Classroom(students,teacher) {
function disp() { //显示所有学生
alert(this.students.join(","));
}
this.students = students;
this.teacher = teacher;
/*注意：Classroom当作普通方法时，disp()可执行;但是作为构造函数，在
*new 对象时，disp()是错误的。
*/
// disp();
}
生成的对象是不能访问构造函数中的私有方法的；
甚至在构造函数中都不能直接调用私有方法，必须通过私有变量或私有方法来调用；
作为普通方法，函数中的私有方法可以直接调用执行，但是如果new就会出错。
