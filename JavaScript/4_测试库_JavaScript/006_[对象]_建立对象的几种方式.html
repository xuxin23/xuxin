建立对象的几种方式
var test = function() {

/*在新建的Object对象上添加属性和方法*/
var obj1 = new Object();
obj1.val = 5;
obj1.click = function() {
alert("this is obj1...");
}

/*用{...}简写方式，结合键值对来定义属性*/
var obj2 = {
val:5,
click:function() {
alert("this is obj2...");
}
}

/*通过new 构造函数的方式*/
var obj3 = new function() {
this.val = 5;
this.click = function() {
alert("this is obj3...");
}
}

obj1.click();
obj2.click();
obj3.click();

}
这里展现了3种创建对象的方式：
1.新建Object,逐步添加属性；2.{...}简写方式结合键值对；3.new 构造函数的方式；
注：前2种方式都是即时创建对象的，第3种方式可以抽离出构造函数，随时new创建对象
