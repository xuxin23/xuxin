用闭包实现curry化
var test = function() {
//addFive现在包含一个接收单一参数的函数，这个函数能求得5加上该参数的和
var addFive = addGenerator(5);
//这里我们可以看到，在传入参数为4时，addFive函数的结果是9
alert(addFive(4) == 9);
}

//数字求和函数的函数生成器
function addGenerator(num) {
//返回一个简单的函数，求两个数字的和，其中第一个数字来自生成器
return function(toAdd) {
return num + toAdd
};
}
Curry化是一种通过把多个参数填充到函数体中，实现将函数转换为一个新的经过简化(使之接受的参数更少)函数的技术。
