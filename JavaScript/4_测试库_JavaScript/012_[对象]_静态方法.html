对象静态方法
var test = function() {
var user = new User("xuxin",23);
/*静态方法必须用User来调用！！和class一样呗*/
var user2 = User.cloneUser(user);
alert(user2.getName() + "-" + user2.getAge());
}

function User(name,age) {
this.name = name;
this.age = age;
this.getName = function() {
return this.name;
};
this.getAge = function() {
return this.age;
};
}

//添加一个User对象的静态方法
User.cloneUser = function(user) {
//创建并返回一个新的用户
return new User (
user.getName(),
user.getAge()
);
};
静态方法，如同java的静态方法一样
形式：Object.method = function() {}
注意：同java一样，只能用Object.method()来调用，而不能用实例化的对象来调用。
