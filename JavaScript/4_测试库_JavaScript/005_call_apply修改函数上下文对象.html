call/apply修改函数上下文对象
var test = function() {
// changeColor("white");//在window对象中调用此函数会失败，因为window对象没有style属性
var title = document.getElementById("title");
changeColor.call(title,"white");//使用call方法
setBodyColor("green");
}

function changeColor(color) {//一个设置上下文对象颜色样式的简单函数
this.style.color = color;
}

function setBodyColor() {//设置body元素颜色的函数
changeColor.apply(document.body,arguments);//apply方法
}
call的第一个参数是上下文对象，其他参数作为原函数的参数
apply的第一个参数是上下文对象，第二个参数为传递给函数的所有参数的数组
