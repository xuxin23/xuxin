公共方法之prototype属性的使用
var test = function() {
var userA = new User("kobe",33);
var userB = new User("wade",30);
/*经验证，输出结果是相同的*/
//函数内部的公共方法
alert(userA.getName() + "-" + userA.getAge());
alert(userB.getName() + "-" + userB.getAge());
//函数的Prototype方法
alert(userA.getPrototypeName() + "-" + userA.getPrototypeAge());
alert(userB.getPrototypeName() + "-" + userB.getPrototypeAge());
}

function User(name,age) {
this.name = name;
this.age = age;
this.getName = function() {
return this.name;
}
this.getAge = function() {
return this.age;
}
}

User.prototype.getPrototypeName = function() {
return this.name;
}

User.prototype.getPrototypeAge = function() {
return this.age;
}
函数内部的公共方法和prototype方法执行的结果是一样的
函数内部的公共方法又叫做特权方法，用来指代那些在查看并处理(对象中)私有变量的同事 允许用户以公共方法的方式访问的方法。
